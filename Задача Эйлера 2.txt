<script>

let perv = 1; //Первое число
let vtor = 1; //Второе число
let tret = perv + vtor; //Сумма двух чисел
let chet_chisla = tret; //Счетчик четных чисел
while(tret <= 4000000) //Пока числа меньше 4 миллионов
{
perv = vtor; //Первому числу присваиваем второе
vtor = tret //Второму числу присваиваем сумма двух чисел
tret = perv + vtor; //Сумма двух чисел

if (tret % 2 == 0 && tret <= 4000000) //Если число делится без остатка и оно меньше 4000000
{
chet_chisla += tret; //Увеличиваем сумму четных чисел
}
}
alert(chet_chisla); //Выводим сумму четных чисел

</script>
